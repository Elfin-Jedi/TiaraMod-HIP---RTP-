rtp_martial_norse = {
	sound = warrior_lodge_ambient_01
	society_ranks_gfx = warriors_ranks

	indestructible = yes
	society_influence = no

	opinion_per_rank_above = 5
	member_score_per_attribute = 1
	member_score_per_rank = 2

		always = yes
	}
	
	can_join_society = {
		martial = 5
		lower_tier_than = KING
		rtp_is_capable_adult = yes
		OR = {
			religion = norse_pagan
			religion = norse_pagan_reformed
			religion_group = pagan_group
			religion_group = dark
		}
		NOR = {
			AND = {
				ai = yes
				is_society_rank_full = {
					society = rtp_martial_norse
					rank = 1
				}
			}
			custom_tooltip = {
				text = "rtp_got_kicked_ct"
				has_character_flag = rtp_kicked_from_martial_society #Bye Felicia!
			}

		OR = {
			religion = norse_pagan

		}
	}

		OR = {
			religion = norse_pagan
			religion = norse_pagan_reformed
			religion_group = pagan_group

		}
	}

	society_rank = {
		level = 1

		startup_limit = 50
		modifier = {
			combat_rating = 5
		}
		powers = {
			rtp_martial_p1_sd

			monthly_character_piety = -0.1
			health = -0.25
		}
		powers = {
			rtp_martial_p3_sd
		}
	}

	society_rank = {
		level = 4
		limit = 5
		startup_limit = 5
		modifier = {
			combat_rating = 20
			martial = 3
			monthly_character_piety = -0.1
			health = -0.5
		}
		powers = {
			rtp_martial_p4_td

		name = currency_name_rtp_martial
